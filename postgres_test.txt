 /*DROP TABLE IF EXISTS categoria, proveedor, producto;*/

  create table categoria(
   id INT PRIMARY KEY,
   nombre VARCHAR(50)
 );
 create table proveedor(
   id INT PRIMARY KEY,
   nombre VARCHAR(50)
 );
 
 create table producto(
   id INT PRIMARY KEY,
   nombre VARCHAR(50),
   cantidad INT,
   fecha_actualizacion DATE,
   idProveedor INT,
   idCategoria INT ,
   FOREIGN KEY(idProveedor) REFERENCES proveedor(id),
   FOREIGN KEY(idCategoria) REFERENCES categoria(id) 
 );

create view view_product 
as select nombre, cantidad, idCategoria from producto  where cantidad > 4  ;
select * from view_product;
create view view_cat 
as select categoria.nombre as nombre_Categoria,  id from categoria  ;
select view_product.nombre, view_product.cantidad, view_cat.nombre_Categoria from view_product inner join view_cat  on view_product.idcategoria = view_cat.id order by view_product.cantidad asc;
